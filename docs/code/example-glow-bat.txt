#!/bin/bash

# Recibimos la respuesta de la IA (por pipe o variable)
input=$(cat)

# Usamos awk para dividir el contenido basado en los delimitadores ```
echo "$input" | awk -v RS='```' '
    NR % 2 == 1 { 
        # Es texto normal (fuera de las comillas)
        print $0 | "glow" 
        close("glow")
    } 
    NR % 2 == 0 { 
        # Es un bloque de código
        # El primer "palabra" suele ser el lenguaje (python, bash, etc.)
        print "```" $0 "```" | "bat --language md --style=numbers,grid"
        close("bat --language md --style=numbers,grid")
    }
'
